/* 4. Define the Bondi metric components as matrix lg
      This is the canonical form with all off-diagonal terms */

lg: matrix(
    /* u-index (0): u coordinate */
    [ - (V/r) * exp(2*b),               /* g\_uu      */
      - exp(2*b),                       /* g\_ur      */
      r\ensuremath{^2} * h\_\ensuremath{\theta}\ensuremath{\theta} * (-U\_\ensuremath{\theta}), /* g\_u\ensuremath{\theta}  (off-diag) */
      r\ensuremath{^2} * h\_\ensuremath{\phi}\ensuremath{\phi}   * (-U\_\ensuremath{\phi}) ],   /* g\_u\ensuremath{\phi}  (off-diag) */

    /* r-index (1): r coordinate */
    [ - exp(2*b),         /* g\_ru = g\_ur */
      0,                  /* g\_rr = 0 (Bondi gauge) */
      0,                  /* g\_r\ensuremath{\theta} = 0 (Bondi gauge) */
      0 ],                /* g\_r\ensuremath{\phi} = 0 (Bondi gauge) */

    /* \ensuremath{\theta}-index (2): \ensuremath{\theta} coordinate */
    [ r\ensuremath{^2} * h\_\ensuremath{\theta}\ensuremath{\theta} * (-U\_\ensuremath{\theta}), /* g\_\ensuremath{\theta}u      */
      0,                  /* g\_\ensuremath{\theta}r = 0  */
      r\ensuremath{^2} * h\_\ensuremath{\theta}\ensuremath{\theta},          /* g\_\ensuremath{\theta}\ensuremath{\theta}      */
      r\ensuremath{^2} * h\_\ensuremath{\theta}\ensuremath{\phi} ],        /* g\_\ensuremath{\theta}\ensuremath{\phi}      */

    /* \ensuremath{\phi}-index (3): \ensuremath{\phi} coordinate */
    [ r\ensuremath{^2} * h\_\ensuremath{\phi}\ensuremath{\phi} * (-U\_\ensuremath{\phi}), /* g\_\ensuremath{\phi}u      */
      0,                  /* g\_\ensuremath{\phi}r = 0  */
      r\ensuremath{^2} * h\_\ensuremath{\theta}\ensuremath{\phi},          /* g\_\ensuremath{\phi}\ensuremath{\theta} = g\_\ensuremath{\theta}\ensuremath{\phi} */
      r\ensuremath{^2} * h\_\ensuremath{\phi}\ensuremath{\phi} ]         /* g\_\ensuremath{\phi}\ensuremath{\phi}      */
)\$
