\href{https://www.youtube.com/watch?v=P_bzpoADnh4&list=PLeFwDGOexoe-0Ay0dqMoJStmy1kn_5V9l&index=18}
{The Pullback of 1-forms}

\begin{equation}
(\mathrm{f}^\ast\omega)_p(v_p)=
\omega_{\mathrm{f}(p)}(\mathrm{f}_\ast v_p)
\in\mathbb{R}
\notag
\end{equation}

\begin{equation}
\mathrm{f}^\ast\omega=
\omega_{\color{blue}j}\left({\mathrm{f}(p)}\right)
\dfrac{\partial\mathrm{f}^{\color{blue}j}}
{\partial x^{\color{blue}i}}\mathrm{d}x^{\color{blue}i}
\notag
\end{equation}

\section{}


Based on Robert Davie Video:
\href{https://youtu.be/7VOQysJHxbE?list=PLeFwDGOexoe-0Ay0dqMoJStmy1kn_5V9l}
{The Pullback of 2-forms}

\textbf{For any point} $p\in M$ \textbf{and vectors}
$v,w\in T_p M$:
\begin{equation}
\left({\mathrm{f}^\ast\omega}\right)_p(v,w)=
\omega\left({\mathrm{f}(p)}\right)
\left({D_p\mathrm{f}(v),D_p\mathrm{f}(w)}\right)
\notag
\end{equation}
